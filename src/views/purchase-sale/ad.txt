import React, { useState, useEffect, useRef } from 'react'
import {
    CButton,
    CCard,
    CCardBody,
    CCol,
    CRow,
    CTable,
    CTableBody,
    CTableDataCell,
    CTableHead,
    CTableHeaderCell,
    CTableRow,
    CModal,
    CModalHeader,
    CModalTitle,
    CModalBody,
    CModalFooter,
    CForm,
    CFormInput,
    CFormLabel,
    CFormSelect,
    CInputGroup,
    CInputGroupText,
    CBadge,
    CListGroup,
    CListGroupItem,
} from '@coreui/react'
import CIcon from '@coreui/icons-react'
import { cilPlus, cilPencil, cilTrash, cilSearch, cilMoney } from '@coreui/icons'
import { purchaseApi, supplierApi, masterApi, saleApi, loanApi } from '../../api/reservationApi'

const PurchaseSale = () => {
    // Current Context
    const [currentYear, setCurrentYear] = useState(new Date().getFullYear())

    // State for data
    const [purchaseData, setPurchaseData] = useState([])
    const [suppliers, setSuppliers] = useState([])
    const [items, setItems] = useState([])
    const [loading, setLoading] = useState(false)

    // State for modals
    const [purchaseModalVisible, setPurchaseModalVisible] = useState(false)
    const [saleModalVisible, setSaleModalVisible] = useState(false)
    const [loanModalVisible, setLoanModalVisible] = useState(false)
    const [isEditing, setIsEditing] = useState(false)
    const [isEditingSale, setIsEditingSale] = useState(false)
    const [isEditingLoan, setIsEditingLoan] = useState(false)
    const [editId, setEditId] = useState(null)
    const [editSaleId, setEditSaleId] = useState(null)
    const [editLoanId, setEditLoanId] = useState(null)

    // Supplier Search State
    const [supplierSearch, setSupplierSearch] = useState('')
    const [showSupplierResults, setShowSupplierResults] = useState(false)
    const searchRef = useRef(null)

    // Purchase Form State
    const [purchaseForm, setPurchaseForm] = useState({
        billDate: new Date().toISOString().split('T')[0],
        billNo: '',
        supplierId: '',
        supplierName: '',
        itemId: '',
        agreementNo: '',
        noOfPacket: '',
        grWt: '',
        cutting: '0',
        netWt: '0.000',
        rate: '',
        amount: '0.00',
        loadingLabour: '0',
        totalCost: '0.00',
        year: currentYear
    })

    // Sale Form State
    const [saleForm, setSaleForm] = useState({
        purchaseId: '',
        saleDt: new Date().toISOString().split('T')[0],
        party: '',
        nikashiPkt: '0',
        tayariPkt: '0',
        charri: '0',
        salePkt: '0',
        saleWt: '',
        rate: '0',
        amount: '0.00',
        unloadingLabour: '0',
        tayaroLabour: '0',
        coldStorageRent: '0',
        newBags: '0',
        sutli: '0',
        pktCollection: '0',
        raffuChipri: '0',
        totalExpOnSales: '0.00',
        netResult: '0.00',
        shortage: '0'
    })

    // Loan Form State
    const [loanForm, setLoanForm] = useState({
        purchaseId: '',
        loanDt: new Date().toISOString().split('T')[0],
        loanAmount: '0',
        interest: '0',
        payRecd: '0',
        netDues: '0.00',
        remarks: ''
    })

    // Fetch initial data
    useEffect(() => {
        fetchInitialData()
        const handleClickOutside = (event) => {
            if (searchRef.current && !searchRef.current.contains(event.target)) {
                setShowSupplierResults(false)
            }
        }
        document.addEventListener('mousedown', handleClickOutside)
        return () => document.removeEventListener('mousedown', handleClickOutside)
    }, [])

    useEffect(() => {
        fetchPurchases()
    }, [currentYear])

    const fetchInitialData = async () => {
        try {
            const [suppliersRes, itemsRes] = await Promise.all([
                supplierApi.getAll(),
                masterApi.getItems()
            ])
            setSuppliers(suppliersRes.data.data)
            setItems(itemsRes.data.data)
        } catch (error) {
            console.error('Error fetching master data:', error)
        }
    }

    const fetchPurchases = async () => {
        setLoading(true)
        try {
            const response = await saleApi.getGrouped(currentYear)
            setPurchaseData(response.data.data)
        } catch (error) {
            console.error('Error fetching purchases:', error)
        } finally {
            setLoading(false)
        }
    }

    const [availableLots, setAvailableLots] = useState([])
    const [selectedLotInfo, setSelectedLotInfo] = useState(null)

    const fetchAvailableLots = async () => {
        try {
            const response = await saleApi.getAvailablePurchases(currentYear)
            setAvailableLots(response.data.data)
        } catch (error) {
            console.error('Error fetching available lots:', error)
        }
    }

    const fetchAutoBillNo = async () => {
        try {
            const response = await purchaseApi.generateBillNo(currentYear)
            setPurchaseForm(prev => ({ ...prev, billNo: response.data.data.billNo }))
        } catch (error) {
            console.error('Error fetching bill no:', error)
        }
    }

    // Auto Calculations Purchase
    useEffect(() => {
        const grWt = parseFloat(purchaseForm.grWt) || 0
        const cutting = parseFloat(purchaseForm.cutting) || 0
        const rate = parseFloat(purchaseForm.rate) || 0
        const loadingLabour = parseFloat(purchaseForm.loadingLabour) || 0

        const netWt = (grWt - cutting).toFixed(3)
        const amount = (parseFloat(netWt) * rate).toFixed(2)
        const totalCost = (parseFloat(amount) + loadingLabour).toFixed(2)

        setPurchaseForm(prev => ({
            ...prev,
            netWt,
            amount,
            totalCost
        }))
    }, [purchaseForm.grWt, purchaseForm.cutting, purchaseForm.rate, purchaseForm.loadingLabour])

    // Auto Calculations Sale
    useEffect(() => {
        const saleWt = parseFloat(saleForm.saleWt) || 0
        const rate = parseFloat(saleForm.rate) || 0
        const amount = (saleWt * rate).toFixed(2)

        const nikashiPkt = parseInt(saleForm.nikashiPkt) || 0
        const tayariPkt = parseInt(saleForm.tayariPkt) || 0

        let unloadingLabour = parseFloat(saleForm.unloadingLabour) || 0
        let tayaroLabour = parseFloat(saleForm.tayaroLabour) || 0
        let coldStorageRent = parseFloat(saleForm.coldStorageRent) || 0

        // Auto-calculate from rates if we have a lot selected and values are 0/empty
        if (selectedLotInfo) {
            // Unloading: per Nikashi packet (default to 0 if not set but we use rate from super admin)
            if (unloadingLabour === 0 || !saleForm.unloadingLabour) {
                unloadingLabour = (nikashiPkt * (selectedLotInfo.itemUnloadingRate || 0))
            }
            // Tayari: per Tayari packet
            if (tayaroLabour === 0 || !saleForm.tayaroLabour) {
                tayaroLabour = (tayariPkt * (selectedLotInfo.itemTaiyariRate || 0))
            }
            // Cold Rent: per Quintal (per weight)
            if (coldStorageRent === 0 || !saleForm.coldStorageRent) {
                coldStorageRent = (saleWt * (selectedLotInfo.itemRentRate || 0))
            }
        }

        const newBags = parseFloat(saleForm.newBags) || 0
        const sutli = parseFloat(saleForm.sutli) || 0
        const pktCollection = parseFloat(saleForm.pktCollection) || 0
        const raffuChipri = parseFloat(saleForm.raffuChipri) || 0

        const totalExp = (unloadingLabour + tayaroLabour + coldStorageRent + newBags + sutli + pktCollection + raffuChipri).toFixed(2)
        const netResult = (parseFloat(amount) - parseFloat(totalExp)).toFixed(2)

        setSaleForm(prev => ({
            ...prev,
            amount,
            unloadingLabour: unloadingLabour.toFixed(2),
            tayaroLabour: tayaroLabour.toFixed(2),
            coldStorageRent: coldStorageRent.toFixed(2),
            totalExpOnSales: totalExp,
            netResult
        }))
    }, [
        saleForm.saleWt, saleForm.rate, saleForm.nikashiPkt, saleForm.tayariPkt,
        saleForm.unloadingLabour, saleForm.tayaroLabour, saleForm.coldStorageRent,
        saleForm.newBags, saleForm.sutli, saleForm.pktCollection, saleForm.raffuChipri,
        selectedLotInfo
    ])

    const handlePurchaseChange = (e) => {
        const { name, value } = e.target
        setPurchaseForm(prev => ({ ...prev, [name]: value }))
    }

    const [salePartySearch, setSalePartySearch] = useState('')
    const [showSalePartyResults, setShowSalePartyResults] = useState(false)
    const salePartySearchRef = useRef(null)

    const handleSaleChange = (e) => {
        const { name, value } = e.target
        setSaleForm(prev => ({ ...prev, [name]: value }))

        if (name === 'purchaseId') {
            const lot = availableLots.find(l => l.id === parseInt(value))
            setSelectedLotInfo(lot)

            // Auto-fill common fields and rates
            if (lot) {
                const isFirstSale = !lot.existingNikashiPkt && !lot.existingTayariPkt

                setSaleForm(prev => ({
                    ...prev,
                    // Common fields from previous sales
                    nikashiPkt: lot.existingNikashiPkt || '0',
                    tayariPkt: lot.existingTayariPkt || '0',
                    charri: lot.existingCharri || '0',
                    newBags: lot.existingNewBags || '0',
                    sutli: lot.existingSutli || '0',
                    pktCollection: lot.existingPktCollection || '0',
                    raffuChipri: lot.existingRaffuChipri || '0',

                    // Rates from Super Admin
                    // We'll calculate these in the useEffect, but we can store the base rates if needed
                }))

                // Also trigger calculations based on these rates in the useEffect
            }
        }
    }

    const selectSaleParty = (supplier) => {
        setSaleForm(prev => ({
            ...prev,
            party: supplier.name
        }))
        setSalePartySearch(supplier.name)
        setShowSalePartyResults(false)
    }

    const selectSupplier = (supplier) => {
        setPurchaseForm(prev => ({
            ...prev,
            supplierId: supplier.id,
            supplierName: supplier.name
        }))
        setSupplierSearch(supplier.name)
        setShowSupplierResults(false)
    }

    const openPurchaseModal = () => {
        setIsEditing(false)
        setEditId(null)
        setPurchaseForm({
            billDate: new Date().toISOString().split('T')[0],
            billNo: '',
            supplierId: '',
            supplierName: '',
            itemId: items.length > 0 ? items[0].id : '',
            agreementNo: '',
            noOfPacket: '',
            grWt: '',
            cutting: '0',
            netWt: '0.000',
            rate: '',
            amount: '0.00',
            loadingLabour: '0',
            totalCost: '0.00',
            year: currentYear
        })
        setSupplierSearch('')
        fetchAutoBillNo()
        setPurchaseModalVisible(true)
    }

    const handleEditPurchase = (purchase) => {
        setIsEditing(true)
        setEditId(purchase.id)
        setPurchaseForm({
            billDate: purchase.billDate,
            billNo: purchase.billNo,
            supplierId: purchase.supplierId,
            supplierName: purchase.supplier?.name || '',
            itemId: purchase.itemId,
            agreementNo: purchase.agreementNo,
            noOfPacket: purchase.noOfPacket,
            grWt: purchase.grWt,
            cutting: purchase.cutting,
            netWt: purchase.netWt,
            rate: purchase.rate,
            amount: purchase.amount,
            loadingLabour: purchase.loadingLabour,
            totalCost: purchase.totalCost,
            year: purchase.year
        })
        setSupplierSearch(purchase.supplier?.name || '')
        setPurchaseModalVisible(true)
    }

    const handleDeletePurchase = async (id) => {
        if (!window.confirm('Are you sure you want to delete this purchase and all its sales?')) return
        try {
            await purchaseApi.delete(id, currentYear)
            fetchPurchases()
        } catch (error) {
            alert(error.response?.data?.message || 'Error deleting purchase')
        }
    }

    const handleSavePurchase = async () => {
        if (!purchaseForm.supplierId || !purchaseForm.agreementNo || !purchaseForm.noOfPacket || !purchaseForm.grWt || !purchaseForm.rate) {
            alert('Please fill all required fields (*)')
            return
        }

        try {
            const { supplierName, ...payload } = purchaseForm
            payload.year = currentYear

            if (isEditing) {
                await purchaseApi.update(editId, payload, currentYear)
            } else {
                await purchaseApi.create(payload)
            }
            setPurchaseModalVisible(false)
            fetchPurchases()
        } catch (error) {
            alert(error.response?.data?.message || 'Error saving purchase')
        }
    }

    const openSaleModal = (purchaseId = '') => {
        setIsEditingSale(false)
        setEditSaleId(null)
        setSalePartySearch('')
        setSelectedLotInfo(null)
        fetchAvailableLots()
        setSaleForm({
            purchaseId: purchaseId,
            saleDt: new Date().toISOString().split('T')[0],
            party: '',
            nikashiPkt: '0',
            tayariPkt: '0',
            charri: '0',
            salePkt: '0',
            saleWt: '',
            rate: '0',
            amount: '0.00',
            unloadingLabour: '0',
            tayaroLabour: '0',
            coldStorageRent: '0',
            newBags: '0',
            sutli: '0',
            pktCollection: '0',
            raffuChipri: '0',
            totalExpOnSales: '0.00',
            netResult: '0.00'
        })
        setSaleModalVisible(true)
    }

    const openLoanModal = (purchaseId = '') => {
        setIsEditingLoan(false)
        setEditLoanId(null)
        fetchAvailableLots()
        setLoanForm({
            purchaseId: purchaseId,
            loanDt: new Date().toISOString().split('T')[0],
            loanAmount: '0',
            interest: '0',
            payRecd: '0',
            netDues: '0.00',
            remarks: ''
        })
        setLoanModalVisible(true)
    }

    const handleLoanChange = (e) => {
        const { name, value } = e.target
        setLoanForm(prev => ({ ...prev, [name]: value }))
    }

    // Auto Calculations Loan
    useEffect(() => {
        const loanAmount = parseFloat(loanForm.loanAmount) || 0
        const interest = parseFloat(loanForm.interest) || 0
        const payRecd = parseFloat(loanForm.payRecd) || 0
        const netDues = (loanAmount + interest - payRecd).toFixed(2)

        setLoanForm(prev => ({ ...prev, netDues }))
    }, [loanForm.loanAmount, loanForm.interest, loanForm.payRecd])

    const handleSaveLoan = async () => {
        if (!loanForm.purchaseId || !loanForm.loanAmount) {
            alert('Please select Lot and enter Loan Amount')
            return
        }

        try {
            if (isEditingLoan) {
                await loanApi.update(editLoanId, loanForm)
            } else {
                await loanApi.create(loanForm)
            }
            setLoanModalVisible(false)
            fetchPurchases()
        } catch (error) {
            alert(error.response?.data?.message || 'Error saving loan')
        }
    }

    const handleEditLoan = (loan) => {
        setIsEditingLoan(true)
        setEditLoanId(loan.id)
        setLoanForm({
            purchaseId: loan.purchaseId,
            loanDt: loan.loanDt,
            loanAmount: loan.loanAmount,
            interest: loan.interest,
            payRecd: loan.payRecd,
            netDues: loan.netDues,
            remarks: loan.remarks || ''
        })
        setLoanModalVisible(true)
    }

    const handleDeleteLoan = async (id) => {
        if (!window.confirm('Are you sure you want to delete this loan entry?')) return
        try {
            await loanApi.delete(id)
            fetchPurchases()
        } catch (error) {
            alert(error.response?.data?.message || 'Error deleting loan')
        }
    }

    const handleEditSale = (sale) => {
        setIsEditingSale(true)
        setEditSaleId(sale.id)
        setSaleForm({
            purchaseId: sale.purchaseId,
            saleDt: sale.saleDt,
            party: sale.party,
            nikashiPkt: sale.nikashiPkt,
            tayariPkt: sale.tayariPkt,
            charri: sale.charri,
            salePkt: sale.salePkt,
            saleWt: sale.saleWt,
            rate: sale.rate,
            amount: sale.amount,
            unloadingLabour: sale.unloadingLabour,
            tayaroLabour: sale.tayaroLabour,
            coldStorageRent: sale.coldStorageRent,
            newBags: sale.newBags,
            sutli: sale.sutli,
            pktCollection: sale.pktCollection,
            raffuChipri: sale.raffuChipri,
            totalExpOnSales: sale.totalExpOnSales,
            netResult: sale.netResult,
            shortage: sale.shortage
        })
        setSaleModalVisible(true)
    }

    const handleDeleteSale = async (id) => {
        if (!window.confirm('Are you sure you want to delete this sale?')) return
        try {
            await saleApi.delete(id)
            fetchPurchases()
        } catch (error) {
            alert(error.response?.data?.message || 'Error deleting sale')
        }
    }

    const handleSaveSale = async () => {
        if (!saleForm.purchaseId || !saleForm.party || !saleForm.saleWt || !saleForm.rate) {
            alert('Please fill all required fields (*)')
            return
        }

        try {
            if (isEditingSale) {
                await saleApi.update(editSaleId, saleForm)
            } else {
                const savedSale = await saleApi.create(saleForm)

                // Automatic Purchase Creation Logic
                try {
                    // Find the supplier object from the name, if exists
                    const salePartySupplier = suppliers.find(s => s.name === saleForm.party)
                    if (salePartySupplier) {
                        const newPurchasePayload = {
                            billDate: saleForm.saleDt, // Use Sale Date as Bill Date
                            supplierId: salePartySupplier.id,
                            itemId: selectedLotInfo?.itemId || (purchaseData.find(p => p.id === parseInt(saleForm.purchaseId))?.itemId),
                            agreementNo: `TRF-${Date.now()}`, // Unique Agreement No
                            noOfPacket: saleForm.salePkt || '0',
                            grWt: saleForm.saleWt || '0',
                            cutting: '0',
                            netWt: saleForm.saleWt || '0',
                            rate: saleForm.rate || '0',
                            amount: saleForm.amount || '0',
                            loadingLabour: '0',
                            totalCost: saleForm.amount || '0',
                            year: currentYear
                        }

                        await purchaseApi.create(newPurchasePayload)
                    }
                } catch (autoPurchaseError) {
                    console.error("Auto Purchase Creation Failed", autoPurchaseError)
                    alert("Auto Purchase Creation Failed: " + (autoPurchaseError.response?.data?.message || autoPurchaseError.message))
                }
            }
            setSaleModalVisible(false)
            fetchPurchases()
        } catch (error) {
            alert(error.response?.data?.message || 'Error saving sale')
        }
    }

    const calculateTotals = () => {
        const totalPackets = purchaseData.reduce((sum, p) => sum + (parseInt(p.noOfPacket) || 0), 0)
        const totalGrWt = purchaseData.reduce((sum, p) => sum + (parseFloat(p.grWt) || 0), 0)
        const totalNetWt = purchaseData.reduce((sum, p) => sum + (parseFloat(p.netWt) || 0), 0)
        const totalAmount = purchaseData.reduce((sum, p) => sum + (parseFloat(p.amount) || 0), 0)
        const totalCost = purchaseData.reduce((sum, p) => sum + (parseFloat(p.totalCost) || 0), 0)

        const allSales = purchaseData.flatMap(p => p.sales || [])
        const totalSaleWt = allSales.reduce((sum, s) => sum + (parseFloat(s.saleWt) || 0), 0)
        const totalSaleAmount = allSales.reduce((sum, s) => sum + (parseFloat(s.amount) || 0), 0)
        const totalNetResult = allSales.reduce((sum, s) => sum + (parseFloat(s.netResult) || 0), 0)
        const allLoans = purchaseData.flatMap(p => p.loans || [])
        const totalLoanAmount = allLoans.reduce((sum, l) => sum + (parseFloat(l.loanAmount) || 0), 0)
        const totalInterest = allLoans.reduce((sum, l) => sum + (parseFloat(l.interest) || 0), 0)
        const totalPayRecd = allLoans.reduce((sum, l) => sum + (parseFloat(l.payRecd) || 0), 0)
        const totalNetDues = allLoans.reduce((sum, l) => sum + (parseFloat(l.netDues) || 0), 0)

        return {
            totalPackets,
            totalGrWt: totalGrWt.toFixed(3),
            totalNetWt: totalNetWt.toFixed(3),
            totalAmount: totalAmount.toFixed(2),
            totalCost: totalCost.toFixed(2),
            totalSaleWt: totalSaleWt.toFixed(3),
            totalSaleAmount: totalSaleAmount.toFixed(2),
            totalNetResult: totalNetResult.toFixed(2),
            totalLoanAmount: totalLoanAmount.toFixed(2),
            totalInterest: totalInterest.toFixed(2),
            totalPayRecd: totalPayRecd.toFixed(2),
            totalNetDues: totalNetDues.toFixed(2)
        }
    }

    const totals = calculateTotals()

    const filteredSuppliers = suppliers.filter(s =>
        s.name.toLowerCase().includes(supplierSearch.toLowerCase()) ||
        (s.mobileNo && s.mobileNo.includes(supplierSearch))
    )

    return (
        <>
            <style>{`
                /* Purchase Blue Theme */
                .purchase-header { background: #0d6efd !important; color: white !important; }
                .purchase-header-light { background: #e7f1ff !important; color: #0d6efd !important; font-weight: 600 !important; }
                .btn-purchase { background: #0d6efd; color: white; border: none; }
                .btn-purchase:hover { background: #0b5ed7; color: white; }
                .badge-purchase { background: #0d6efd !important; }
                .text-purchase { color: #0d6efd !important; }
                
                /* Sale Green Theme */
                .sale-header { background: #198754 !important; color: white !important; }
                .sale-header-light { background: #e8f5e9 !important; color: #198754 !important; font-weight: 600 !important; }
                .btn-sale { background: #198754; color: white; border: none; }
                .btn-sale:hover { background: #157347; color: white; }
                .badge-sale { background: #198754 !important; }
                .text-sale { color: #198754 !important; }

                /* Loan Orange/Red Theme */
                .loan-header { background: #fd7e14 !important; color: white !important; }
                .loan-header-light { background: #fff3cd !important; color: #fd7e14 !important; font-weight: 600 !important; }
                .btn-loan { background: #fd7e14; color: white; border: none; }
                .btn-loan:hover { background: #e96b02; color: white; }
                .badge-loan { background: #fd7e14 !important; }
                .text-loan { color: #fd7e14 !important; }

                .table-container { background: white; }
                .total-row { background: #f8f9fa !important; font-weight: 700; border-top: 2px solid #0d6efd !important; }
                .calculated-field { background-color: #f1f3f5 !important; font-weight: 600; color: #495057; }
                .search-results {
                    position: absolute; width: 100%; z-index: 1050; max-height: 200px;
                    overflow-y: auto; box-shadow: 0 4px 6px -1px rgba(0,0,0,0.1);
                    border: 1px solid #dee2e6; background: white;
                }
                .form-section-title {
                    font-size: 11px; font-weight: 700; color: #0d6efd;
                    text-transform: uppercase; margin-bottom: 15px; padding-bottom: 5px;
                    border-bottom: 2px solid #e7f1ff;
                }
            `}</style>

            <CRow className="mb-3">
                <CCol md={4} className="d-flex align-items-center gap-2">
                    <CFormLabel className="mb-0 fw-bold">Financial Year:</CFormLabel>
                    <CFormSelect size="sm" style={{ width: '120px' }} value={currentYear} onChange={(e) => setCurrentYear(e.target.value)}>
                        {[2024, 2025, 2026].map(y => <option key={y} value={y}>{y}</option>)}
                    </CFormSelect>
                </CCol>
            </CRow>

            <CCard className="border-0 shadow-sm mb-4">
                <CCardBody className="p-0">
                    <div className="table-container" style={{ width: '100%', overflowX: 'auto' }}>
                        <CTable bordered hover className="text-center align-middle mb-0" style={{ fontSize: '0.73rem', minWidth: '3200px' }}>
                            <CTableHead>
                                <CTableRow>
                                    <CTableHeaderCell colSpan={15} className="purchase-header text-uppercase py-2">
                                        <div className="d-flex align-items-center justify-content-between px-2">
                                            <CButton size="sm" className="btn-light text-primary fw-bold" onClick={openPurchaseModal}>
                                                <CIcon icon={cilPlus} className="me-1" /> Add Purchase
                                            </CButton>
                                            <span className="fs-6 fw-semibold">PURCHASE DETAILS - {currentYear}</span>

                                <CFormLabel className="small fw-bold">Summary</CFormLabel>
                                <div className="p-2 border rounded bg-light small">
                                    <div className="d-flex justify-content-between"><span>Amount:</span> <b>₹ {saleForm.amount}</b></div>
                                    <div className="d-flex justify-content-between"><span>Total Exp:</span> <b>₹ {saleForm.totalExpOnSales}</b></div>
                                    <div className="d-flex justify-content-between border-top mt-1 pt-1 fw-bold text-success"><span>Net Result:</span> <b>₹ {saleForm.netResult}</b></div>
                                </div>
                            </CCol>
                        </CRow>
                    </CForm>
                </CModalBody>
                <CModalFooter className="bg-light p-2"><CButton color="secondary" size="sm" onClick={() => setSaleModalVisible(false)}>Close</CButton><CButton color="primary" size="sm" onClick={handleSaveSale}>Save Sale</CButton></CModalFooter>
            </CModal>
            {/* Loan Modal */}
            <CModal visible={loanModalVisible} onClose={() => setLoanModalVisible(false)} size="lg" backdrop="static">
                <CModalHeader className="bg-warning text-dark p-3"><CModalTitle className="fs-6">{isEditingLoan ? 'Edit Loan Entry' : 'Add New Loan Entry'}</CModalTitle></CModalHeader>
                <CModalBody className="p-4 bg-white">
                    <CForm>
                        <CRow className="mb-3">
                            <CCol md={6}><CFormLabel className="small fw-bold">Select Purchase Lot *</CFormLabel>
                                <CFormSelect name="purchaseId" size="sm" value={loanForm.purchaseId} onChange={handleLoanChange} disabled={isEditingLoan}>
                                    <option value="">Select Lot...</option>
                                    {availableLots.map(p => <option key={p.id} value={p.id}>Lot {p.lotNo} - {p.supplier}</option>)}
                                </CFormSelect>
                            </CCol>
                            <CCol md={6}><CFormLabel className="small fw-bold">Loan Date</CFormLabel><CFormInput type="date" name="loanDt" size="sm" value={loanForm.loanDt} onChange={handleLoanChange} /></CCol>
                        </CRow>
                        <CRow className="mb-3">
                            <CCol md={4}><CFormLabel className="small fw-bold">Loan Amount (₹) *</CFormLabel><CFormInput type="number" name="loanAmount" size="sm" value={loanForm.loanAmount} onChange={handleLoanChange} /></CCol>
                            <CCol md={4}><CFormLabel className="small fw-bold">Interest (₹)</CFormLabel><CFormInput type="number" name="interest" size="sm" value={loanForm.interest} onChange={handleLoanChange} /></CCol>
                            <CCol md={4}><CFormLabel className="small fw-bold">Payment Recd (₹)</CFormLabel><CFormInput type="number" name="payRecd" size="sm" value={loanForm.payRecd} onChange={handleLoanChange} /></CCol>
                        </CRow>
                        <CRow className="mb-3">
                            <CCol md={12}><CFormLabel className="small fw-bold">Remarks</CFormLabel><CFormInput name="remarks" size="sm" value={loanForm.remarks} onChange={handleLoanChange} placeholder="Reason for loan..." /></CCol>
                        </CRow>
                        <div className="p-2 border rounded bg-light text-end">
                            <span className="fw-bold text-danger">NET DUES/PAYABLE: ₹ {loanForm.netDues}</span>
                        </div>
                    </CForm>
                </CModalBody>
                <CModalFooter className="bg-light p-2"><CButton color="secondary" size="sm" onClick={() => setLoanModalVisible(false)}>Close</CButton><CButton color="warning" size="sm" className="text-dark fw-bold" onClick={handleSaveLoan}>Save Loan</CButton></CModalFooter>
            </CModal>
        </>
    )
}

export default PurchaseSale
